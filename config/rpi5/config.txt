# Raspberry Pi 5 config.txt for the headless LED driver board.
#
# Pi 5's RP1 chip has uniform SPI controllers (unlike the BCM2711 aux SPI),
# so both SPI0 and SPI1 toggle CS reliably at end-of-transfer --- no
# userspace XLAT workaround needed.

arm_64bit=1
start_file=start4.elf
fixup_file=fixup4.dat

enable_uart=1
disable_splash=1
disable_overscan=1
arm_boost=1

dtparam=spi=on

# SPI0 for :input_left chain, SPI1 for :main chain. cs0_pin= redirects
# kernel CS off our XLAT pins onto unused GPIOs, freeing GPIO 8/18 for
# Circuits.GPIO-managed manual XLAT in Chain.ex. Matches the Pi 4
# approach --- kernel-CS-as-XLAT on Pi 5 works *usually* but has
# occasional glitches (2026-04-17 bench), so we take control of the
# XLAT edge with our own busy-wait timing instead.
#
# (The :main connector is silk-screened "SPI3" on the power-distribution
# board but the node boards are resoldered to SPI1 pins. See
# docs/build_v2_hardware.md.)
dtoverlay=spi0-1cs,cs0_pin=26
dtoverlay=spi1-1cs-pi5,cs0_pin=25

dtoverlay=dwc2
dtoverlay=ramoops

# Audio disabled to avoid pin conflicts with SPI1
dtparam=audio=off
