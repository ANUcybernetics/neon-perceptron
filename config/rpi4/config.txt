# Raspberry Pi 4 config.txt for the stock nerves_system_rpi4 build.
#
# This config targets a headless Pi 4B used as the LED driver board:
# no display output, no camera, no DSI panel --- just SPI for driving
# daisy-chained TLC5947 boards.

arm_64bit=1
start_file=start4.elf
fixup_file=fixup4.dat

enable_uart=1
disable_splash=1

# Enable SPI. Under nerves_system_rpi4 this exposes /dev/spidev0.0 and
# /dev/spidev0.1 on the default pins (GPIO 7-11).
dtparam=spi=on

# SPI0 (GPIO 10/11/8) for :input_left chain.
dtoverlay=spi0-1cs

# SPI3 (GPIO 2/3/0) for :main chain. Uses the CN4 connector on the power
# distribution board (silk-screened "SPI3"). SPI3 is a BCM2711 aux SPI
# controller --- CE0 on GPIO 0 may need manual XLAT via Chain.xlat_gpio.
dtoverlay=spi3-1cs

# Audio disabled to avoid GPIO pin conflicts with SPI buses.
dtparam=audio=off
