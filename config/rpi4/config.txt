# Raspberry Pi 4 config.txt for the stock nerves_system_rpi4 build.
#
# This config targets a headless Pi 4B used as the LED driver board:
# no display output, no camera, no DSI panel --- just SPI for driving
# daisy-chained TLC5947 boards.

arm_64bit=1
start_file=start4.elf
fixup_file=fixup4.dat

enable_uart=1
disable_splash=1

# Enable SPI. Under nerves_system_rpi4 this exposes /dev/spidev0.0 and
# /dev/spidev0.1 on the default pins (GPIO 7-11).
dtparam=spi=on

# SPI0 (GPIO 10/11, manual XLAT on GPIO 8) for :input_left chain.
# cs0_pin=26 redirects kernel CE0 to unused GPIO 26, freeing GPIO 8 for
# manual XLAT via Chain.xlat_gpio. Kernel CE doesn't pulse reliably enough
# to latch the TLC5947 on Pi 4, so all chains use manual XLAT.
dtoverlay=spi0-1cs,cs0_pin=26

# SPI1 (GPIO 20/21, manual XLAT on GPIO 18) for :main chain. Uses the CN4
# connector on the power distribution board (silk-screened "SPI3") but the
# node boards have been resoldered to route SPI1 pins. cs0_pin=25 frees
# GPIO 18 for manual XLAT. BCM2711 aux SPI --- manual XLAT required.
dtoverlay=spi1-1cs,cs0_pin=25

# Audio disabled to avoid GPIO pin conflicts with SPI buses.
dtparam=audio=off
