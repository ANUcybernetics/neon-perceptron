# Raspberry Pi 4 config.txt for the stock nerves_system_rpi4 build.
#
# This config targets a headless Pi 4B used as the LED driver board:
# no display output, no camera, no DSI panel --- just SPI for driving
# daisy-chained TLC5947 boards.

arm_64bit=1
start_file=start4.elf
fixup_file=fixup4.dat

enable_uart=1
disable_splash=1

# Enable SPI. Under nerves_system_rpi4 this exposes /dev/spidev0.0 and
# /dev/spidev0.1 on the default pins (GPIO 7-11).
dtparam=spi=on

# Per-chain 1-CS overlays: SPI0 CE0 (GPIO 8) and SPI1 CE0 (GPIO 18)
# drive XLAT for their respective TLC5947 daisy chains. See TASK-16.
dtoverlay=spi0-1cs
dtoverlay=spi1-1cs

# Audio and i2s disabled to free GPIO 18-21 for SPI1.
dtparam=audio=off
